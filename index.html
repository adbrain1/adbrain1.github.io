    function loadWeather(lat,lon,elId,label){
    fetch(`https://api.open-meteo.com/v1/forecast?latitude=${lat}&longitude=${lon}&current_weather=true`)
    .then(res=>res.json())
    .then(data=>{
    const temp=Math.round(data.current_weather.temperature);
    const code=data.current_weather.weathercode;
    let icon="☀️";
    if([1,2,3].includes(code)) icon="⛅";
    if([45,48].includes(code)) icon="🌫️";
    if([51,53,55,61,63,65,80].includes(code)) icon="🌧️";
    if([71,73,75,85].includes(code)) icon="❄️";
    if([95,96,99].includes(code)) icon="⛈️";
    document.getElementById(elId).textContent=`${label} ${icon} ${temp}°C`;
    });
    }
